select * from dba_tablespace_usage_metrics;
col user_bytes format 999999999999
select * from dba_data_files; 
select SEGMENT_NAME, SEGMENT_TYPE, TABLESPACE_NAME, MB,
                empty_blocks, avg_space, num_rows, avg_row_len,chain_cnt,last_analyzed
  from (select ds.owner,
segment_name,
               segment_type,
               ds.tablespace_name,
               ds.bytes / 1024 / 1024 MB,
               rank() over(order by ds.bytes desc) srank,
               dt.empty_blocks,dt.avg_space,dt.num_rows,dt.avg_row_len,dt.chain_cnt,to_char(last_analyzed,'YYYYMMDD') last_analyzed
          from dba_segments ds,dba_tables dt
         where ds.tablespace_name= 'DG_INDEX'
           and ds.segment_name = dt.table_name(+)
           and ds.owner = dt.owner(+)
           )
where MB >=20
order by MB desc;

select SEGMENT_NAME, SEGMENT_TYPE, TABLESPACE_NAME, MB,
                empty_blocks, avg_space, num_rows, avg_row_len,chain_cnt,last_analyzed
  from (select ds.owner,
segment_name,
               segment_type,
               ds.tablespace_name,
               ds.bytes / 1024 / 1024 MB,
               rank() over(order by ds.bytes desc) srank,
               dt.empty_blocks,dt.avg_space,dt.num_rows,dt.avg_row_len,dt.chain_cnt,to_char(last_analyzed,'YYYYMMDD') last_analyzed
          from dba_segments ds,dba_tables dt
         where ds.tablespace_name= 'DG_DATA'
           and ds.segment_name = dt.table_name(+)
           and ds.owner = dt.owner(+)
           )
where MB >=20
order by MB desc;

select fs.tablespace_name "Tablespace",
       (df.totalspace - fs.freespace) "Used MB",
       fs.freespace "Free MB",
       df.totalspace "Total MB",
       round(100 * (fs.freespace / df.totalspace)) "Pct. Free"
  from (select tablespace_name, round(sum(bytes) / 1048576) TotalSpace
          from dba_data_files
         group by tablespace_name) df,
       (select tablespace_name, round(sum(bytes) / 1048576) FreeSpace
          from dba_free_space
         group by tablespace_name) fs
where df.tablespace_name = fs.tablespace_name(+);
